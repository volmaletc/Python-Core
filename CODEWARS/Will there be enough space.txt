def enough(cap, on, wait):
    if on + wait == cap:
        return 0
    elif on + wait < cap:
        return 0
    elif on > wait and on + wait !=cap:
        return wait - (cap - on)
    elif on < wait and on + wait !=cap:
        return wait - (cap - on)
    elif on == wait and on + wait !=cap:
        return wait - (cap - on)
